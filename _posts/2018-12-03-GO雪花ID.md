 ### 程式 ###

 ````go
package snowflake

import (
	"sync"
	"time"
)

const (
	nodeBits  uint8 = 10        				// 節點 ID 的位數 (0-1023)
	indexBits  uint8 = 12             			// Index 的位數 (0-4095)
	indexMax  int64 = -1 ^ (-1 << indexBits)    // Index的最大值 用於檢測益出 (4095)
	timeShift uint8 = nodeBits + indexBits   	// 時間戳向左偏移量 (22位)
	nodeShift uint8 = indexBits  				// 節點 ID 向左的偏移量 (10位)
	epoch int64 = 1543809608000					// 2018-12-03 12:00:08
)

type ID int64

type NodeStruct struct {
	mutex sync.Mutex	// 添加互斥鎖 保證併發安全性
	timestamp int64		// 時間戳 部分
	nodeType int64		// Node Type 部分
	index int64 		// Index ID 部分
}

func Node(nodeType int64) (*NodeStruct) {
	// 返回 node 實例指針
	return &NodeStruct{
		timestamp: 0,
		nodeType: nodeType,
		index: 0,
	}
}

func (ns *NodeStruct) Generate() int64 {
	// 生成時加鎖
	ns.mutex.Lock()
	// 完成時解鎖
	defer ns.mutex.Unlock()

	// 得到當前時間戳
	now := time.Now().UnixNano() / 1e6
	if ns.timestamp == now{
		ns.index ++
		// 當前 index 用完
		if ns.index > indexMax {
			// 等待本毫秒结束
			for now <= ns.timestamp {
				now = time.Now().UnixNano() / 1e6
			}
		}
	} else {
		ns.index = 0
		ns.timestamp = now
	}

	// 位移運算生成 ID
	result := int64( (now - epoch) << timeShift | (ns.nodeType << nodeShift) | (ns.index) )
	return result
}

func Inverse (nodeId ID) (*NodeStruct){
	// 返回 node 實例指針
	return &NodeStruct{
		timestamp: int64((nodeId >> timeShift) + ID(epoch) ),
		nodeType: int64((nodeId >> nodeShift) - (nodeId >> timeShift << timeShift >> nodeShift)),
		index: int64( (nodeId) -(nodeId) >> indexBits << indexBits ),
	}
}
 ```
